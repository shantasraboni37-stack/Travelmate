// ═══════════════════════════════════════════════════════════
//  NAV ITEM — Single drawer menu row
// ═══════════════════════════════════════════════════════════
class _NavItem extends StatefulWidget {
  final IconData icon;
  final IconData iconFilled;
  final String label;
  final String route;
  final bool isActive;
  final String? badge;       // optional pill badge e.g. "3"
  final VoidCallback onTap;

  const _NavItem({
    required this.icon,
    required this.iconFilled,
    required this.label,
    required this.route,
    required this.isActive,
    required this.onTap,
    this.badge,
  });

  @override
  State<_NavItem> createState() => _NavItemState();
}

class _NavItemState extends State<_NavItem> {
  bool _hovered = false;

  @override
  Widget build(BuildContext context) {
    final active = widget.isActive;
    final hovered = _hovered && !active;

    return MouseRegion(
      onEnter: (_) => setState(() => _hovered = true),
      onExit:  (_) => setState(() => _hovered = false),
      child: GestureDetector(
        onTap: widget.onTap,
        child: AnimatedContainer(
          duration: const Duration(milliseconds: 160),
          margin: const EdgeInsets.symmetric(vertical: 2),
          padding: const EdgeInsets.symmetric(
              horizontal: 14, vertical: 12),
          decoration: BoxDecoration(
            color: active
                ? _primarySoft
                : hovered
                    ? _surfaceLow
                    : Colors.transparent,
            borderRadius: BorderRadius.circular(14),
            border: Border.all(
              color: active ? _primaryFixed : Colors.transparent,
              width: 1.5,
            ),
          ),
          child: Row(
            children: [
              // Icon
              Icon(
                active ? widget.iconFilled : widget.icon,
                size: 22,
                color: active ? _primary : _textSub,
              ),
              const SizedBox(width: 14),

              // Label
              Expanded(
                child: Text(
                  widget.label,
                  style: GoogleFonts.manrope(
                    fontSize: 14,
                    fontWeight: active
                        ? FontWeight.w700
                        : FontWeight.w500,
                    color: active ? _primary : _dark,
                  ),
                ),
              ),

              // Badge or chevron
              if (widget.badge != null)
                Container(
                  padding: const EdgeInsets.symmetric(
                      horizontal: 8, vertical: 3),
                  decoration: BoxDecoration(
                    color: _primary,
                    borderRadius: BorderRadius.circular(999),
                    boxShadow: [
                      BoxShadow(
                        color: _primary.withOpacity(0.28),
                        blurRadius: 8,
                        offset: const Offset(0, 2),
                      ),
                    ],
                  ),
                  child: Text(
                    widget.badge!,
                    style: GoogleFonts.manrope(
                      fontSize: 11,
                      fontWeight: FontWeight.w800,
                      color: _onPrimary,
                    ),
                  ),
                )
              else if (active)
                const Icon(Icons.chevron_right_rounded,
                    size: 18, color: _primary),
            ],
          ),
        ),
      ),
    );
  }
}

                style: GoogleFonts.manrope(
                  fontSize: 14,
                  fontWeight: FontWeight.w500,
                  color: _dark,
                ),

              onTap: () => onChanged(!value),

                  duration: const Duration(milliseconds: 220),
                  curve: Curves.easeInOut,

                      ? Alignment.centerRight
                      : Alignment.centerLeft,
                  child: Container(
                    margin: const EdgeInsets.all(3),
                    width: 18,

// ═══════════════════════════════════════════════════════════
//  LOGOUT BUTTON
// ═══════════════════════════════════════════════════════════
class _LogoutButton extends StatefulWidget {
  final VoidCallback onTap;
  const _LogoutButton({required this.onTap});

  @override
  State<_LogoutButton> createState() => _LogoutButtonState();
}

class _LogoutButtonState extends State<_LogoutButton> {
  bool _pressed = false;

  @override
  Widget build(BuildContext context) {
    return GestureDetector(
      onTapDown: (_) => setState(() => _pressed = true),
      onTapUp:   (_) { setState(() => _pressed = false); widget.onTap(); },
      onTapCancel: () => setState(() => _pressed = false),
      child: AnimatedScale(
        scale: _pressed ? 0.97 : 1.0,
        duration: const Duration(milliseconds: 90),
        child: Container(
          width: double.infinity,
          padding: const EdgeInsets.symmetric(
              horizontal: 14, vertical: 13),
          decoration: BoxDecoration(
            color: const Color(0xFFFFF0F2),
            borderRadius: BorderRadius.circular(14),
            border: Border.all(
                color: const Color(0xFFFFDADB), width: 1.5),
          ),
          child: Row(
            children: [
              const Icon(Icons.logout_rounded,
                  size: 20, color: _primary),
              const SizedBox(width: 14),
              Text(
                'Logout',
                style: GoogleFonts.manrope(
                  fontSize: 14,
                  fontWeight: FontWeight.w700,
                  color: _primary,
                ),
              ),
            ],
          ),
        ),
      ),
    );
  }
}

// ═══════════════════════════════════════════════════════════
class _TravelBottomNav extends StatelessWidget {
  final int activeIndex;
  final ValueChanged<int> onTap;

  const _TravelBottomNav({
    required this.activeIndex,
    required this.onTap,
  });

  static const List<_NavTab> _tabs = [
    _NavTab(icon: Icons.group_outlined,       filled: Icons.group_rounded,          label: 'Buddies'),
    _NavTab(icon: Icons.chat_bubble_outline,  filled: Icons.chat_bubble_rounded,    label: 'Chats'),
    _NavTab(icon: Icons.explore_outlined,     filled: Icons.explore_rounded,        label: 'Community'),
    _NavTab(icon: Icons.card_travel_outlined, filled: Icons.card_travel_rounded,    label: 'Packages'),
    _NavTab(icon: Icons.person_outline,       filled: Icons.person_rounded,         label: 'Profile'),
  ];

  @override
  Widget build(BuildContext context) {
    return Container(
      decoration: BoxDecoration(
        color: _surfaceCard.withOpacity(0.95),
        borderRadius: const BorderRadius.only(
          topLeft:  Radius.circular(26),
          topRight: Radius.circular(26),
        ),
        boxShadow: const [
          BoxShadow(
            color: Color(0x0D000000),
            blurRadius: 24,
            offset: Offset(0, -4),
          ),
        ],
        border: Border(
          top: BorderSide(color: _outlineVariant, width: 1),
        ),
      ),
      child: SafeArea(
        top: false,
        child: Padding(
          padding: const EdgeInsets.symmetric(horizontal: 8, vertical: 8),
          child: Row(
            mainAxisAlignment: MainAxisAlignment.spaceAround,
            children: List.generate(_tabs.length, (i) {
              final tab   = _tabs[i];
              final active = i == activeIndex;
              return GestureDetector(
                onTap: () => onTap(i),
                behavior: HitTestBehavior.opaque,
                child: AnimatedContainer(
                  duration: const Duration(milliseconds: 200),
                  padding: EdgeInsets.symmetric(
                    horizontal: active ? 16 : 12,
                    vertical: 8,
                  ),
                  decoration: BoxDecoration(
                    color: active ? _primarySoft : Colors.transparent,
                    borderRadius: BorderRadius.circular(14),
                  ),
                  child: Column(
                    mainAxisSize: MainAxisSize.min,
                    children: [
                      Icon(
                        active ? tab.filled : tab.icon,
                        size: 24,
                        color: active ? _primary : _textSub,
                      ),
                      const SizedBox(height: 3),
                      Text(
                        tab.label,
                        style: GoogleFonts.manrope(
                          fontSize: 10,
                          fontWeight: active
                              ? FontWeight.w700
                              : FontWeight.w500,
                          color: active ? _primary : _textSub,
                        ),
                      ),
                    ],
                  ),
                ),
              );
            }),
          ),
        ),
      ),
    );
  }
}

class _BuddyCard extends StatelessWidget {
  final String name;
  final String dest;
  final String dates;
  final String type;
  final String imageUrl;

  const _BuddyCard({
    required this.name,
    required this.dest,
    required this.dates,
    required this.type,
    required this.imageUrl,
  });

  @override
  Widget build(BuildContext context) {
    return Container(
      margin: const EdgeInsets.only(bottom: 14),
      padding: const EdgeInsets.all(16),
      decoration: BoxDecoration(
        color: _surfaceCard,
        borderRadius: BorderRadius.circular(18),
        border: Border.all(color: _outlineVariant, width: 1.5),
        boxShadow: const [
          BoxShadow(
            color: Color(0x08000000),
            blurRadius: 12,
            offset: Offset(0, 3),
          ),
        ],
      ),
      child: Row(
        children: [
          // Avatar
          ClipOval(
            child: Image.network(
              imageUrl,
              width: 58, height: 58, fit: BoxFit.cover,
              errorBuilder: (_, __, ___) => Container(
                width: 58, height: 58,
                color: _primaryFixed,
                child: const Icon(Icons.person_rounded,
                    color: _primary, size: 30),
              ),
            ),
          ),
          const SizedBox(width: 14),

          // Info
          Expanded(
            child: Column(
              crossAxisAlignment: CrossAxisAlignment.start,
              children: [
                Text(
                  name,
                  style: GoogleFonts.manrope(
                    fontSize: 15, fontWeight: FontWeight.w800, color: _dark,
                  ),
                ),
                const SizedBox(height: 3),
                Row(
                  children: [
                    const Icon(Icons.location_on_outlined,
                        size: 13, color: _primary),
                    const SizedBox(width: 3),
                    Expanded(
                      child: Text(
                        dest,
                        style: GoogleFonts.manrope(
                          fontSize: 12, color: _textSub, fontWeight: FontWeight.w500,
                        ),
                        overflow: TextOverflow.ellipsis,
                      ),
                    ),
                  ],
                ),
                const SizedBox(height: 6),
                Wrap(
                  spacing: 6,
                  children: [
                    _Chip(icon: Icons.calendar_month_outlined, label: dates),
                    _Chip(icon: Icons.group_outlined,          label: type),
                  ],
                ),
              ],
            ),
          ),

          // Connect button
          const SizedBox(width: 8),
          OutlinedButton(
